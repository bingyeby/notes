```html
<cover-view class="bg-wrap"
            bindtap="onDictTap"
            data-dict-name="{{item.name}}">
    <cover-image class="bg-img" src="../../images/bg.jpg" bindtap="onDictTap"
                    data-dict-name="{{item.name}}"/>

<cover-view class="text-wrap"
```
